CLASS AND INSTANCE METHODS


            class Animal
              def eat
                puts 'I am eating'
              end
            end

            animal = Animal.new
            animal.eat # prints "I am eating"

INHERITANCE - SUBCLASSING


            class Person < Animal
              def talk
                puts 'blah blah'
              end
            end

            class Cat < Animal
              def meow
                puts "meow"
              end
            end

            person = Person.new
            person.talk # prints "blah blah"
            person.eat  # prints "I am eating"

Modules as mix-ins

To share behaviour where inheritance is not viable


            module SwimAbility
              def swim
                puts "I'm swimming"
              end
            end

            class Fish < Animal
              include SwimAbility
            end

            class Person
              include SwimAbility
            end

            person.swim # prints "I'm swimming"

METHODS VISIBILITY

public, protected


            class Woman
              def initialize(name, age)
                @name, @age = name, age
              end

              def older?(other)
                (age <=> other.age) == 1 ? true : false
              end

              protected
              attr_accessor :age # a woman will never tell you her age
            end

            mother   = Woman.new('Jane', 44)
            daughter = Woman.new('Jill', 12)
            mother.older? daughter # => true

METHODS VISIBILITY

public, private


            class Woman
              def initialize(name, age)
                @name, @age = name, age
              end

              def older?(other)
                (age <=> other.age) == 1 ? true : false
              end

              private
              attr_accessor :age # a woman will never tell anyone her age
            end

            mother   = Woman.new('Jane', 44)
            daughter = Woman.new('Jill', 12)
            mother.older? daughter # raises a NoMethodError exception

Symbols

symbols are similar to strings, but their object_id never changes


              'abc'.object_id == 'abc'.object_id # => false
              :abc.object_id == :abc.object_id   # => true

Bang Methods

bang methods are "dangerous" versions or regular methods. Often used when method changes the object itself


              string = 'satanic message'
              string.reverse
              string # => "satanic message"

              string.reverse!
              string # => "egassem cinatas"

Flow Control

If


            if true
              puts "true"
            end

            puts "true" if 3 > 1
            puts "true" if 0
            puts "true" if ''

            puts "this never happens" if false
            puts "this never happens" if nil

            puts "true" if !false
            puts "true" if !nil

Else


            if 3 > 5
              puts "3 is greater than 5"
            else
              puts "3 is not greater than 5"
            end

Elsif


            if 3 > 5
              puts "3 is greater than 5"
            elsif 3 > 2
              puts "3 is greater than 2"
            else
              "3 is not greater than 5 and not greater than 2"
            end

Unless


            puts "true" if !(3 < 1)
            # same as
            puts "true" unless 3 < 1

Each

While


            n = 0
            while n < 10
              puts n
              n += 1
            end

Until


            n = 1
            until n >= 10
              puts n
              n += 1
            end

Loop & sleep


            n = 1
            loop do
              puts n
              n += 1
              sleep 1 # number of seconds
            end

DUCK TYPING (polymorphism)

If it quacks like a duck then it's a duck


            def make_speak(animal)
              if animal.respond_to? :meow
                animal.meow
              elsif animal.respond_to? :talk
                animal.talk
              end
            end

            animal = Cat.new
            make_speak(animal)
