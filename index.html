CLASS AND INSTANCE METHODS


            class Animal
              def eat
                puts 'I am eating'
              end
            end

            animal = Animal.new
            animal.eat # => I am eating

INHERITANCE - SUBCLASSING


            class Person < Animal
              def talk
                puts 'blah blah'
              end
            end

            class Cat < Animal
              def meow
                puts "meow"
              end
            end

Modules as mix-ins

To share behaviour where inheritance is not viable


            module SwimAbility
              def swim
                puts "I'm swimming"
              end
            end

            class Fish < Animal
              include SwimAbility
            end

            class Person
              include SwimAbility
            end

cats cannot swim

DUCK TYPING (polymorphism)

If it quacks like a duck then it's a duck


            def make_speak(animal)
              if animal.respond_to? :meow
                animal.meow
              elsif animal.respond_to? :talk
                animal.talk
              end
            end

            animal = Cat.new
            make_speak(animal)
